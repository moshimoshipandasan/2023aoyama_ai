- ピラミッドを作るプログラム例
```
level = 0
side_length = 0
def build_pyramid(height: number):
    global side_length, level
    agent.set_item(SANDSTONE, 1, 1)
    agent.set_assist(PLACE_ON_MOVE, True)
    agent.set_assist(DESTROY_OBSTACLES, True)
    while level <= height - 1:
        side_length = height - level
        for index in range(4):
            if index == 3:
                agent.move(FORWARD, side_length - 1)
                agent.turn(LEFT_TURN)
            else:
                agent.move(FORWARD, side_length)
                agent.turn(LEFT_TURN)
        # 次の層に移動する
        agent.move(FORWARD, 1)
        agent.move(UP, 1)
        level += 2
    if height % 2 == 0:
        agent.move(UP, 1)


def on_on_chat():
    build_pyramid(10)
player.on_chat("ピラミッドを作る", on_on_chat)
```
